\section{Recall}
\begin{itemize}
    \itemsep 1.0em
    \item {probability = wanted outcomes/total possible outcomes}
    \item {P(A and B) = P(A) x P(B)}
    \item {P(A or B) = P(A) + P(B) - P(A and B)}
    \item {P(not A) = 1 - P(A)}
\end{itemize}
